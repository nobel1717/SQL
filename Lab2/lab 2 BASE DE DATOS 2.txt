 CREATE DATABASE LABORATORIO2
USE LABORATORIO2

--CREACION DE TABLA PROPIETARIO
CREATE TABLE PROPIETARIO (
CODPROPIETARIO VARCHAR(50) PRIMARY KEY NOT NULL,
CORREO VARCHAR(50) DEFAULT '@gmail.com',
NOMBRE VARCHAR(50),
APELLIDO VARCHAR(50),
TELEFONO VARCHAR(50),
GENERO VARCHAR(50), 
DIRECCION VARCHAR(50)
)

--CREACION DE TABLA POLICIA
CREATE TABLE POLICIA (
[CODIGO OFICIAL] VARCHAR(50) PRIMARY KEY NOT NULL,
NOMBRE VARCHAR(50),
APELLIDO VARCHAR(50),
DIRECCION VARCHAR(50),
TELEFONO VARCHAR(50),
RANGO VARCHAR(50) DEFAULT 'CADETE'
)



--CREACION DE TABLA VEHICULO
CREATE TABLE VEHICULO(
PLACA VARCHAR (50) PRIMARY KEY NOT NULL,
NUM_MOTOR VARCHAR (50) NOT NULL,
MARCA VARCHAR (50),
COLOR VARCHAR (50),
TIPO_POLIZA VARCHAR (50),
MODELO VARCHAR (50),
CLASE VARCHAR(50),
CODPROPIETARIO VARCHAR (50) FOREIGN KEY REFERENCES PROPIETARIO (CODPROPIETARIO),

)


--CFREACION DE TABLA MULTAS

CREATE TABLE MULTAS (
[CODIGO DE MULTA] VARCHAR(50),
PLACA VARCHAR(50),
DESCRIPCION VARCHAR(50),
IMPORTE VARCHAR(50) DEFAULT '50',
FECHA DATETIME,
LUGAR VARCHAR(50),
HORA TIME NOT NULL,
CONSTRAINT PK_MULTAS PRIMARY KEY ([CODIGO DE MULTA], PLACA),
CONSTRAINT FK_MULTAS_VEHICULO FOREIGN KEY (PLACA) REFERENCES VEHICULO (PLACA),
CONSTRAINT FK_MULTAS_POLICIA FOREIGN KEY ([CODIGO OFICIAL]) REFERENCES POLICIA ([CODIGO OFICIAL])
)


--CREACION DE TABLA ATTT
CREATE TABLE ENTIDAD(
RUC VARCHAR (50) PRIMARY KEY NOT NULL,
NOMBRE VARCHAR (50)NOT NULL,
DIRECCION VARCHAR (50),
TELEFONO VARCHAR (50),
)
